jQuery.requireModule()

Simple Test
        function loadModule(name) {
            jQuery.requireModule(name)
                .done(function(){
                    $(arguments).each(function() {
                        $(ele).append("<div>" + this + " loaded</div>");
                    });
                })
                .fail(function() {
                    $(arguments).each(function() {
                        $(ele).append("<div>" + this + " failed</div>");
                    });
                });
        }
